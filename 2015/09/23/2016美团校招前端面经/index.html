类和继承

javascript类

javascript是一种面向原型的语言，函数是javascript的一等公民。具体参考：前端学习笔记之类和继承

ES5类

function Animal(name) {
    this.name = name;
}
Animal.prototype.call = function() {
    console.log("I'm a", this.name);
}
var lion = new Animal("Lion");
lion.call();
lion.name = 'new lion';
lion.call();

ES6类

class Animal {
    constructor(name) {
        this.name = name;
    }
    call() {
        console.log("I'm a", this.name);
    }
    set name(name) {
        this.name = name;
    }
    get name() {
        return this.name;
    }
}
var lion = new Animal("lion");
lion.call();
lion.name = 'new lion';
lion.call();

javascript继承

javascript采用原型链方式实现继承，每个类都有一个prototype/__proto__属性指向其构造函数，prototype.__proto__指向其继承的父类的构造函数，默认每个类的父类都是Object类。

> Function.prototype
< function () {}
> Function.prototype.__proto__
< Object {}
> Function.__proto__
< function () {}
> Object.prototype
< Object {}
> Object.__proto__
< function() {}

ES5继承

Child.prototype = Object.create(Parent.prototype);
Child.prototype = new Parent();
Child.prototype.__proto__ = Parent.prototype;

ES6继承

class Child extends Parent {
    constructor(name) {
        super(name);
        // this name
    }
    call() {
        super.call();
    }
}

事件绑定

DOM实现满足兼容性的事件绑定

function addEvent(ele, type, handler) {
    if(ele.addEventListener) {
        ele.addEventListener(type, handler, false);
    } else if(ele.attachEvent) {
        ele.attachEvent("on"+type, handler);
    } else {
        ele["on"+type] = handler;
    }
}

对象实现事件绑定

var obj = {
    on: function(type, handler) {
        if(!this[type]) {
            this[type] = [];
        }
        this[type].push(handler);
    },
    trigger: function(type) {
        if(this[type]) {
            this[type].forEach(function(fn){
                fn();
            });
        }
    }
};

获取对象属性：点操作和中括号操作

var obj = {name: 'sally'}，用点操作的时候，后面需要制定一个属性名称，例如obj.name，但是如果想要动态添加属性时，var age = "age"，obj.age就是不正确的，这时age会被当做一个数字，而不是变量；使用中括号操作时，括号里面可以是变量也可以是字符串，比较随意，中括号会计算第一次生成字符串访问。其实，在执行obj.name操作时，最终也会按照obj["name"]去执行的。
